# install 
```bash
npm i github:hardianto-cpu/telegram-sticker
```
# usage
```js
// common js
const { getSticker } = require('telegram-sticker');
const build = async () => {
const sticker = await getSicker('anime');
console.log(sticker);
}
build();

// es6
import { getSticker } from 'telegram-sticker';
const build = async () => {
const sticker = await getSicker('anime');
console.log(sticker);
}
build();
```
